**Fix Part B — Update `useAuth.jsx` to also store refresh_token and handle email-based initials fallback:**

**Replace the entire file `src/hooks/useAuth.jsx` with this:**

```jsx
import { useState, createContext, useContext } from "react";

const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(() => {
    const token = localStorage.getItem("token");
    const userData = localStorage.getItem("user");
    if (token && userData) {
      try { return JSON.parse(userData); }
      catch { return { token }; }
    }
    return null;
  });

  const login = (userData) => {
    setUser(userData);
    localStorage.setItem("user", JSON.stringify(userData));
    localStorage.setItem("token", userData.token || userData.access_token);
    if (userData.refresh_token) {
      localStorage.setItem("refresh_token", userData.refresh_token);
    }
  };

  const logout = () => {
    setUser(null);
    localStorage.removeItem("token");
    localStorage.removeItem("refresh_token");
    localStorage.removeItem("user");
  };

  return (
    <AuthContext.Provider value={{ user, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => useContext(AuthContext);
```

### PROBLEM 4 — Token refresh endpoint called with wrong HTTP method and wrong parameter format

**File:** `src/services/api.js`

**Problem:** The refresh token interceptor calls:
```js
await axios.post("/api/auth/refresh", null, { params: { refresh_token: refreshToken } })
```
But the backend endpoint `POST /api/auth/refresh` expects `refresh_token` as a **query string parameter** (`def refresh_token(refresh_token: str, ...)`). Using `params` with axios on a POST actually does send it as a query string, which works — BUT when the interceptor retries the original request it uses `axios(originalRequest)` instead of `api(originalRequest)`. This means the base URL isn't applied correctly when retrying. Additionally, the interceptor doesn't clear the failed state properly if the user object exists but the token expired.

**Also:** `SettingsPage.jsx` has a hardcoded email address (`spkotaaa@gmail.com`) instead of reading from the auth context.

**Replace the entire file `src/services/api.js` with this:**

```js
import axios from "axios";

const api = axios.create({
  baseURL: "/api",
  headers: {
    "Content-Type": "application/json",
  },
});

api.interceptors.request.use((config) => {
  const token = localStorage.getItem("token");
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

api.interceptors.response.use(
  (response) => response,
  async (error) => {
    const originalRequest = error.config;
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true;
      const refreshToken = localStorage.getItem("refresh_token");
      if (refreshToken) {
        try {
          // Use the api instance so baseURL is correct; send as query param
          const response = await api.post(`/auth/refresh?refresh_token=${encodeURIComponent(refreshToken)}`);
          const { access_token, refresh_token: newRefreshToken } = response.data;
          localStorage.setItem("token", access_token);
          localStorage.setItem("refresh_token", newRefreshToken);
          // Update user object in localStorage with new token
          try {
            const userStr = localStorage.getItem("user");
            if (userStr) {
              const userObj = JSON.parse(userStr);
              userObj.token = access_token;
              localStorage.setItem("user", JSON.stringify(userObj));
            }
          } catch (_) {}
          originalRequest.headers.Authorization = `Bearer ${access_token}`;
          return api(originalRequest);
        } catch (refreshError) {
          localStorage.removeItem("token");
          localStorage.removeItem("refresh_token");
          localStorage.removeItem("user");
          window.location.href = "/login";
        }
      } else {
        localStorage.removeItem("token");
        localStorage.removeItem("user");
        window.location.href = "/login";
      }
    }
    return Promise.reject(error);
  }
);

### PROBLEM 6 — File upload polling always marks documents as "ready" regardless of actual backend status

### PROBLEM 7 — Backend: FAISS index is stored as a single flat file and is NOT per-user, causing all users' document chunks to be mixed together

**File:** `backend/services/embedding_service.py`

**Problem:** The FAISS index stores embeddings from ALL users' documents in a single global flat file. When `search()` filters by `document_ids`, it correctly filters by `doc_id` in metadata — but only if ALL documents from ALL users are in the same index. This works at small scale but is a design flaw that causes incorrect RAG results if two users upload documents with the same content. The metadata filter by `doc_id` is the correct safeguard, but the `save_index()` method is called while the lock is held in `add_to_index()` but NOT in `_load_index()` after checking file size. This can cause a race condition.

**Fix:** The metadata filter is the right approach. The main fix needed is ensuring the `save_index()` lock protection is consistent. Additionally, the `search()` method uses `1.0 / (1.0 + distance)` for scoring but uses a hardcoded threshold of `0.3` which may be too aggressive, silently dropping relevant results.

**Replace the entire file `backend/services/embedding_service.py` with this:**

```python
from sentence_transformers import SentenceTransformer
import faiss
import numpy as np
import os
import pickle
import threading
import structlog
from config import settings
from rag.chunking import clean_text

logger = structlog.get_logger()

class EmbeddingService:
    _model = None
    _model_lock = threading.Lock()

    def __init__(self):
        self.model = self._get_model()
        self.index_path = settings.FAISS_INDEX_PATH
        self.metadata_path = settings.FAISS_INDEX_PATH + ".metadata"
        self.lock = threading.Lock()
        self._load_index()
        print(f"[FAISS] Startup complete. ntotal={self.index.ntotal}, metadata_count={len(self.metadata)}")

    @classmethod
    def _get_model(cls):
        if cls._model is None:
            with cls._model_lock:
                if cls._model is None:
                    cls._model = SentenceTransformer(settings.EMBEDDING_MODEL)
        return cls._model

    def _load_index(self):
        with self.lock:
            try:
                if os.path.exists(self.index_path):
                    if os.path.getsize(self.index_path) < 1024:
                        print(f"[FAISS] Index file too small ({os.path.getsize(self.index_path)} bytes) — reinitializing")
                        self.index = faiss.IndexFlatL2(384)
                        self.metadata = []
                        self._save_index_unsafe()
                        return
                    self.index = faiss.read_index(self.index_path)
                    if os.path.exists(self.metadata_path):
                        with open(self.metadata_path, "rb") as f:
                            self.metadata = pickle.load(f)
                    else:
                        self.metadata = []
                else:
                    self.index = faiss.IndexFlatL2(384)
                    self.metadata = []
                    self._save_index_unsafe()
            except Exception as exc:
                logger.error("faiss_load_failed", error=str(exc))
                self.index = faiss.IndexFlatL2(384)
                self.metadata = []

    def generate_embeddings(self, texts: list) -> np.ndarray:
        embeddings = self.model.encode(
            texts,
            batch_size=32,
            show_progress_bar=False,
            normalize_embeddings=True,
        )
        return np.array(embeddings, dtype="float32")

    def add_to_index(self, texts: list, doc_metadata: dict):
        if not texts:
            return

        normalized_texts = [clean_text(text) for text in texts]
        if not any(normalized_texts):
            return

        embeddings = self.generate_embeddings(normalized_texts)

        with self.lock:
            try:
                self.index.add(embeddings)
                for text in normalized_texts:
                    self.metadata.append({
                        "content": text,
                        "text": text,
                        "filename": doc_metadata.get("filename", "unknown"),
                        "doc_id": doc_metadata.get("doc_id"),
                        "user_id": doc_metadata.get("user_id"),
                    })
                self._save_index_unsafe()
                print(f"[FAISS] Document indexed. Chunks added={len(normalized_texts)}, ntotal now={self.index.ntotal}")
            except Exception as exc:
                logger.error("faiss_add_failed", error=str(exc))

    def search(self, query: str, k: int = 5, document_ids: list = None):
        if self.index.ntotal == 0:
            return []
        if not query or not query.strip():
            return []

        query_embedding = self.generate_embeddings([clean_text(query)])

        with self.lock:
            try:
                top_k = min(max(1, k * 3), self.index.ntotal)  # retrieve more, then filter
                distances, indices = self.index.search(query_embedding, top_k)

                results = []
                for rank, idx in enumerate(indices[0]):
                    if idx != -1 and idx < len(self.metadata):
                        meta = self.metadata[idx]
                        if document_ids and meta.get("doc_id") not in document_ids:
                            continue
                        result = dict(meta)
                        distance = float(distances[0][rank])
                        score = 1.0 / (1.0 + distance)
                        # Lowered threshold from 0.3 to 0.15 to avoid silently dropping relevant results
                        if score < 0.15:
                            logger.info("document_filtered_low_score", score=score)
                            continue
                        result["score"] = score
                        results.append(result)
                        if len(results) >= k:
                            break
                return results
            except Exception as exc:
                logger.error("faiss_search_failed", error=str(exc))
                return []

    def _save_index_unsafe(self):
        """Save index — caller must hold self.lock."""
        try:
            faiss.write_index(self.index, self.index_path)
            with open(self.metadata_path, "wb") as f:
                pickle.dump(self.metadata, f)
        except Exception as exc:
            logger.error("faiss_save_failed", error=str(exc))

    def save_index(self):
        """Public save — acquires lock."""
        with self.lock:
            self._save_index_unsafe()

embedding_service = EmbeddingService()
```

                document.status = "error"
                db.commit()

from sqlalchemy.orm import declarative_base, sessionmaker
from config import settings

SQLALCHEMY_DATABASE_URL = settings.DATABASE_URL

engine_args = {}
if SQLALCHEMY_DATABASE_URL.startswith("sqlite"):
    engine_args["connect_args"] = {"check_same_thread": False}
else:
    engine_args.update({
        "pool_size": 20,
        "max_overflow": 10,
        "pool_timeout": 30,
        "pool_recycle": 1800,
    })

engine = create_engine(SQLALCHEMY_DATABASE_URL, **engine_args)
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

Base = declarative_base()

def get_db():
    db = SessionLocal()

```python
from google import genai
import json
import structlog
from config import settings
from schemas.chat_schema import ChatResponse, Source

logger = structlog.get_logger()

class RAGService:
    def __init__(self):
        self.client = None
        if settings.GEMINI_API_KEY:
            self.client = genai.Client(api_key=settings.GEMINI_API_KEY)
        self.model_name = settings.GEMINI_MODEL

    def _format_structured_text(self, answer: str, reasoning: str, sources: list) -> str:
        source_lines = "\n".join(sources) if sources else "No sources provided"
        return (
            f"Answer:\n{answer}\n\n"
            f"Reasoning:\n{reasoning}\n\n"
            f"Sources:\n{source_lines}"
        )

    def _build_sources(self, context_chunks: list) -> list:
        sources = []
        seen_files = set()
        for chunk in context_chunks:
            filename = chunk.get("filename", "unknown")
            if filename not in seen_files:
                sources.append(
                    Source(
                        filename=filename,
                        content=(chunk.get("content") or chunk.get("text") or "")[:200] + "...",
                    )
                )
                seen_files.add(filename)
        return sources

    async def generate_response(self, query: str, context_chunks: list) -> ChatResponse:
        print(f"[RAG] Called. model={self.model_name}, api_key_set={bool(self.client)}, context_chunks={len(context_chunks) if context_chunks else 0}")
        context_chunks = context_chunks or []

        if self.client is None:
            answer = "The chat service is currently unavailable."
            reasoning = "GEMINI_API_KEY is not configured, so model generation cannot be executed."
            structured = self._format_structured_text(answer, reasoning, [])
            return ChatResponse(
                summary=answer,
                key_points=[reasoning],
                explanation=structured,
                sources=[]
            )

        # Basic prompt injection detection
        injection_keywords = ["ignore previous instructions", "system prompt", "you are now", "instead of"]
        if any(keyword in query.lower() for keyword in injection_keywords):
            answer = "Request blocked by safety policy."
            reasoning = "Potential prompt injection pattern was detected in the query."
            structured = self._format_structured_text(answer, reasoning, [])
            return ChatResponse(
                summary=answer,
                key_points=[reasoning],
                explanation=structured,
                sources=[]
            )

        context_text = "\n\n".join([(chunk.get("content") or chunk.get("text") or "") for chunk in context_chunks])
        source_refs = [f"{s.filename}: {s.content}" for s in self._build_sources(context_chunks)]

        system_mode = "RAG" if context_chunks else "CHAT"

        prompt = f"""
        You are an AI assistant for DecisionLens.
        Be strict, precise, and instruction-following.

        Mode: {system_mode}
        - If mode is RAG, only use the provided context.
        - If information is missing in RAG mode, say "I do not know".
        - Do not fabricate information.
        - Prefer short, direct answers unless more detail is required.
        - Explain reasoning clearly and only from allowed information.

        Context:
        {context_text}

        User Query:
        {query}

        Return ONLY a valid JSON object with these exact keys:
        {{
          "answer": "direct response to the query",
          "reasoning": "why this answer was reached",
          "pros": ["advantage 1", "advantage 2"],
          "cons": ["disadvantage 1", "disadvantage 2"],
          "risks": ["risk 1", "risk 2"],
          "recommendation": "optional suggested next step"
        }}
        Do NOT wrap in markdown. Do NOT add any text outside the JSON object.
        """

        try:
            response = await self.client.aio.models.generate_content(
                model=self.model_name,
                contents=prompt,
                config={
                    "response_mime_type": "application/json",
                }
            )
            cleaned = response.text.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("```")[1] if "```" in cleaned else cleaned
                cleaned = cleaned.removeprefix("json").strip()
            try:
                data = json.loads(cleaned)
            except json.JSONDecodeError as je:
                logger.error(f"[RAG] JSON parse failed: {je}. Raw response: {cleaned[:200]}")
                data = {
                    "answer": "Unable to generate a response at this time.",
                    "reasoning": "LLM call failed — check server logs for details.",
                    "pros": [],
                    "cons": [],
                    "risks": [],
                    "recommendation": ""
                }

            answer = (data.get("answer") or "").strip() or "I do not know based on the available information."
            reasoning = (data.get("reasoning") or "").strip() or "No additional reasoning provided."
            pros = data.get("pros") or []
            cons = data.get("cons") or []
            risks = data.get("risks") or []
            recommendation = data.get("recommendation") or ""

        except Exception as e:
            logger.error(f"[RAG] Gemini API call failed: {e}", exc_info=True)
            print(f"[RAG] ERROR: {e}")
            answer = "Unable to generate a response at this time."
            reasoning = "LLM call failed — check server logs for details."
            pros = []
            cons = []
            risks = []
            recommendation = ""
            data = {"answer": answer, "reasoning": reasoning}

        sources = self._build_sources(context_chunks)
        structured = self._format_structured_text(
            data.get("answer", answer),
            data.get("reasoning", reasoning),
            source_refs,
        )

        key_points = []
        key_points.extend(["PRO: " + p for p in pros])
        key_points.extend(["CON: " + c for c in cons])
        key_points.extend(["RISK: " + r for r in risks])
        if recommendation:
            key_points.append("RECOMMENDATION: " + recommendation)
        if not key_points:
            key_points.append(reasoning)

        return ChatResponse(
            summary=answer,
            key_points=key_points,
            explanation=structured,
            sources=sources
        )

rag_service = RAGService()

import { useState, useEffect } from "react";
import { getDocuments } from "../services/api";
import { Card } from "../components/ui/Card";
import { Loader } from "../components/ui/Loader";
import { FileText, AlertCircle } from "lucide-react";

export function ExplanationPage() {
  const [documents, setDocuments] = useState([]);
  const [loading, setLoading] = useState(true);

          <span className="text-sm font-medium">{error}</span>
        </div>
      )}

          <div className="w-16 h-16 rounded-full bg-[#D6E4FF] flex items-center justify-center text-primary mx-auto mb-4">

                <FileText size={20} />
              </div>
              <div className="flex-1 min-w-0">
                <p className="font-bold text-[#1E293B] truncate">{doc.filename}</p>
